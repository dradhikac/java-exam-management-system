CREATE DATABASE dsatm_exam_system;

USE dsatm_exam_system;

CREATE TABLE students (
    id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(100) NOT NULL,
    usn VARCHAR(50) UNIQUE,
    temp_roll_no VARCHAR(50) UNIQUE,
    semester INT NOT NULL,
    department VARCHAR(50) NOT NULL
);

CREATE TABLE subjects (
    id INT AUTO_INCREMENT PRIMARY KEY,

    subject_code VARCHAR(20) NOT NULL UNIQUE,
    subject_name VARCHAR(100) NOT NULL,

    semester INT NOT NULL,

    category_code VARCHAR(10) NOT NULL,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_subject_category
        FOREIGN KEY (category_code)
        REFERENCES course_categories(category_code)
);

CREATE TABLE internal_assessments (
    id INT AUTO_INCREMENT PRIMARY KEY,

    student_id INT NOT NULL,
    subject_id INT NOT NULL,

    cie_theory INT,
    cie_lab INT,
    cie_total INT,

    eligible_for_see BOOLEAN NOT NULL,
    eligibility_reason VARCHAR(100),

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    UNIQUE KEY uk_student_subject (student_id, subject_id)
);

CREATE TABLE attendance (
    id INT AUTO_INCREMENT PRIMARY KEY,

    student_id INT NOT NULL,
    subject_id INT NOT NULL,

    theory_attendance_percent DOUBLE,
    lab_attendance_percent DOUBLE,
    attendance_percent DOUBLE,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    UNIQUE KEY uk_student_subject (student_id, subject_id)
);

CREATE TABLE exam_schedule (
    id INT AUTO_INCREMENT PRIMARY KEY,
    subject_id INT NOT NULL,
    semester INT NOT NULL,
    session VARCHAR(50) NOT NULL,
    exam_date DATE NOT NULL,attendance
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    UNIQUE(subject_id, session)
);
